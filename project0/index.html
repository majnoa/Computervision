#vizualize a batch of images

def visualize(batch, nrows=1, ncols=8, labels=None):
    X,y = batch
    if labels is None or len(labels) == 0: #if labels are not provided as list of strings find them
        labels = text_labels(y)
    
    X = X.squeeze(1)

    fig, axes = plt.subplots(nrows, ncols, figsize=(ncols*2, nrows*2))
    axes = axes.flatten()

    for i in range(nrows * ncols):
        if i >= len(X):  # stop if fewer images than grid slots
            axes[i].axis("off")
            continue
        axes[i].imshow(X[i], cmap="viridis")
        axes[i].set_title(labels[i], fontsize=9)
        axes[i].axis("off")

    plt.tight_layout()
    plt.show()

#softmax implementation
def softmax(X):
    exp_X = torch.exp(X)
    sum_exp_X = torch.sum(exp_X, dim=1, keepdim=True)
    return exp_X / sum_exp_X
In [70]:
#cross-entropy loss implementation
def cross_entropy(probs, y):
    # probs: [batch, C], y: [batch]
    p = probs[torch.arange(len(probs)), y].clamp(min=1e-12)
    return -torch.log(p).mean()

#accuracy implementation
def accuracy(y_hat, y):
    preds = y_hat.argmax(dim=1)
    return (preds == y).float().mean().item()

#build the softmax model
class SoftmaxModel(torch.nn.Module): #use nn.module here
    def __init__(self, num_inputs, num_outputs, sigma=0.01):
        super().__init__()
        self.linear = nn.Linear(num_inputs, num_outputs) #first linear layer -> flattens the image into a vector
        #initialize weights of the linear layer
        nn.init.normal_(self.linear.weight, mean=0, std=sigma)
        nn.init.zeros_(self.linear.bias)
    
    def forward(self, X):
        X = X.reshape(X.shape[0], -1)    # flatten the input tensor
        logits = self.linear(X)          # shape [batch, num_outputs]
        return softmax(logits)           # return probabilities

    def loss(self, y_hat, y):
        return cross_entropy(y_hat, y)

def train(model, train_loader, test_loader, epochs=10, lr=0.1):
    optimizer = torch.optim.SGD(model.parameters(), lr=lr)

    history = {
        "train_loss": [], 
        "train_acc": [], 
        "test_loss": [], 
        "test_acc": []
    }

    for epoch in range(epochs):
        # --- Training ---
        model.train()
        total_loss, total_acc, n = 0.0, 0.0, 0
        for X, y in train_loader:
            y_hat = model(X)
            loss = model.loss(y_hat, y)

            optimizer.zero_grad()
            loss.backward()
            optimizer.step()

            batch_size = y.size(0)
            total_loss += loss.item() * batch_size
            total_acc  += accuracy(y_hat, y) * batch_size
            n += batch_size

        epoch_loss = total_loss / n
        epoch_acc  = total_acc / n
        history["train_loss"].append(epoch_loss)
        history["train_acc"].append(epoch_acc)

        # --- Validation/Test ---
        model.eval()
        with torch.no_grad():
            val_loss, val_acc, m = 0.0, 0.0, 0
            for X, y in test_loader:
                y_hat = model(X)
                loss = model.loss(y_hat, y)

                batch_size = y.size(0)
                val_loss += loss.item() * batch_size
                val_acc  += accuracy(y_hat, y) * batch_size
                m += batch_size

            val_loss /= m
            val_acc  /= m
            history["test_loss"].append(val_loss)
            history["test_acc"].append(val_acc)

        print(f"Epoch {epoch+1}: "
              f"train loss={epoch_loss:.4f}, train acc={epoch_acc:.3f}, "
              f"val loss={val_loss:.4f}, val acc={val_acc:.3f}")

    return history

def plot_history(history):
    plt.figure(figsize=(8,5))
    plt.plot(history["train_loss"], label="train loss")
    plt.plot(history["test_loss"], label="val loss")
    plt.plot(history["train_acc"], label="train acc")
    plt.plot(history["test_acc"], label="val acc")
    plt.xlabel("epoch")
    plt.ylabel("value")
    plt.legend()
    plt.title("Training history")
    plt.grid(True)
    plt.show()

2.1 Try to use different learning rates, plot how validation loss changes.¶

The learning rate selection directly influenced how the validation loss function behaved during training. The validation loss decreased steadily with a learning rate of 0.1 before reaching a stable minimum value which indicated successful convergence and strong generalization performance. The validation loss showed unstable behavior when using a high learning rate of 0.5 because parameter updates exceeded the optimal point. The validation loss decreased at a very slow rate under a learning rate of 0.01 before reaching a high plateau which indicated underfitting and poor learning efficiency.

2.2 Try to use different batch sizes, how validation and training loss changes?¶

The training and validation loss behavior showed clear changes when different batch sizes were used for the model. The training loss decreased gradually with a small batch size of 15 yet stayed at high levels while validation loss showed significant fluctuations between epochs. The small batch size produces unstable gradient estimates which result in this pattern of behavior. The training and validation loss curves became more stable when the batch size increased to 60 because the optimization process became more consistent and showed less fluctuation. The training loss dropped quickly to lower values when using a batch size of 200 while the validation loss followed the training loss pattern in a smooth descent. The results indicate that bigger batch sizes produce better gradient stability and faster training but extremely large batch sizes might decrease model generalization performance relative to batches of medium size.

Question 3: Explain the overflow and underflow problems in softmax computation, $\hat y_j = \frac{\exp(o_j)}{\sum_k \exp(o_k)}$. One possible solution to this problem is to subtract $\bar{o} \stackrel{\textrm{def}}{=} \max_k o_k$ from all entries, explain how this potentially can solve the overflow problem. Is this solution perfect or you think this solution may cause some other problems (e.g., NaN)?¶

The overflow and underflow problem arises because the exponential function grows and decays very quickly. If a logit $o_j$ is very large (e.g., $o_j = 1000$), then $\exp(o_j)$ can exceed the representable floating-point range. This results in infinite values and causes the computation to break down. Similarly, a logit $o_j$ is very negative (e.g., $o_j = -1000$), then $\exp(o_j)$ becomes so small that it underflows to zero.
